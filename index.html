WAREHOUSE IMS
Connect your Supabase database to sync inventory across all devices in real time.
1
Go to supabase.com → create a free project
2
In your project → Settings → API → copy Project URL and anon public key
3
Go to SQL Editor → paste and run the setup SQL below
4
Paste your credentials here and click Connect
create table items ( id uuid primary key default gen_random_uuid(), name text not null, sku text unique not null, barcode text, unit text default 'pcs', thresh int default 5, sold boolean default false, locations jsonb default '[]', created_at timestamptz default now() ); create table activity ( id uuid primary key default gen_random_uuid(), type text, item_id uuid references items(id), item_name text, location text, qty int, reason text, notes text, ts timestamptz default now() ); alter table items enable row level security; alter table activity enable row level security; create policy "public" on items for all using (true) with check (true); create policy "public" on activity for all using (true) with check (true);
